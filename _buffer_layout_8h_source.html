466 template <typename T>
467 bool SharedDataStream<T>::BufferLayout::init(size_t wordSize, size_t maxReaders) {
468  // Make sure parameters are not too large to store.
469  if (wordSize > std::numeric_limits<decltype(Header::wordSize)>::max()) {
470  logger::acsdkError(logger::LogEntry(TAG, "initFailed")
471  .d("reason", "wordSizeTooLarge")
472  .d("wordSize", wordSize)
473  .d("wordSizeLimit", std::numeric_limits<decltype(Header::wordSize)>::max()));
474  return false;
475  }
476  if (maxReaders > std::numeric_limits<decltype(Header::maxReaders)>::max()) {
477  logger::acsdkError(logger::LogEntry(TAG, "initFailed")
478  .d("reason", "maxReadersTooLarge")
479  .d("maxReaders", maxReaders)
480  .d("maxReadersLimit", std::numeric_limits<decltype(Header::maxReaders)>::max()));
481  return false;
482  }
483 
484  // Pre-calculate some pointers and sizes that are frequently accessed.
485  calculateAndCacheConstants(wordSize, maxReaders);
486 
487  // Default construction of the Header.
488  auto header = new (getHeader()) Header;
489 
490  // Default construction of the reader arrays.
491  size_t id;
492  for (id = 0; id < maxReaders; ++id) {
493  new (m_readerEnabledArray + id) AtomicBool;
494  new (m_readerCursorArray + id) AtomicIndex;
495  new (m_readerCloseIndexArray + id) AtomicIndex;
496  }
497 
498  // Header field initialization.
499  header->magic = MAGIC_NUMBER;
500  header->version = VERSION;
501  header->traitsNameHash = stableHash(T::traitsName);
502  header->wordSize = wordSize;
503  header->maxReaders = maxReaders;
504  header->isWriterEnabled = false;
505  header->hasWriterBeenClosed = false;
506  header->writeStartCursor = 0;
507  header->writeEndCursor = 0;
508  header->oldestUnconsumedCursor = 0;
509  header->referenceCount = 1;
510 
511  // Reader arrays initialization.
512  for (id = 0; id < maxReaders; ++id) {
513  m_readerEnabledArray[id] = false;
514  m_readerCursorArray[id] = 0;
515  m_readerCloseIndexArray[id] = 0;
516  }
517 
518  return true;
519 }

521 template <typename T>
522 bool SharedDataStream<T>::BufferLayout::attach() {
523  // Verify compatibility.
524  auto header = getHeader();
525  if (header->magic != MAGIC_NUMBER) {
526  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
527  .d("reason", "magicNumberMismatch")
528  .d("magicNumber", header->magic)
529  .d("expectedMagicNumber", std::to_string(MAGIC_NUMBER)));
530  return false;
531  }
532  if (header->version != VERSION) {
533  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
534  .d("reason", "incompatibleVersion")
535  .d("version", header->version)
536  .d("expectedVersion", std::to_string(VERSION)));
537  return false;
538  }
539  if (header->traitsNameHash != stableHash(T::traitsName)) {
540  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
541  .d("reason", "traitsNameHashMismatch")
542  .d("hash", header->traitsNameHash)
543  .d("expectedHash", stableHash(T::traitsName)));
544  return false;
545  }
546 
547  // Attach.
548  std::lock_guard<Mutex> lock(header->attachMutex);
549  if (0 == header->referenceCount) {
550  logger::acsdkError(logger::LogEntry(TAG, "attachFailed").d("reason", "zeroUsers"));
551  return false;
552  }
553  if (std::numeric_limits<decltype(header->referenceCount)>::max() == header->referenceCount) {
554  logger::acsdkError(logger::LogEntry(TAG, "attachFailed")
555  .d("reason", "bufferMaxUsersExceeded")
556  .d("numUsers", header->referenceCount)
557  .d("maxNumUsers", std::numeric_limits<decltype(header->referenceCount)>::max()));
558  return false;
559  }
560  ++header->referenceCount;
561 
562  // Pre-calculate some pointers and sizes that are frequently accessed.
563  calculateAndCacheConstants(header->wordSize, header->maxReaders);
564 
565  return true;
566 }

568 template <typename T>
569 void SharedDataStream<T>::BufferLayout::detach() {
570  if (!isAttached()) {
571  return;
572  }
573 
574  auto header = getHeader();
575  std::lock_guard<Mutex> lock(header->attachMutex);
576  --header->referenceCount;
577  if (header->referenceCount > 0) {
578  return;
579  }
580 
581  // Destruction of reader arrays.
582  for (size_t id = 0; id < header->maxReaders; ++id) {
583  m_readerCloseIndexArray[id].~AtomicIndex();
584  m_readerCursorArray[id].~AtomicIndex();
585  m_readerEnabledArray[id].~AtomicBool();
586  }
587 
588  // Destruction of the Header.
589  header->~Header();
590 }

624 template <typename T>
625 void SharedDataStream<T>::BufferLayout::updateOldestUnconsumedCursorLocked() {
626  auto header = getHeader();
627 
628  // Note: as an optimization, we could skip this function if Writer policy is nonblockable (ACSDK-251).
629 
630  // The only barrier to a blocking writer overrunning a reader is oldestUnconsumedCursor, so we have to be careful
631  // not to ever move it ahead of any readers. The loop below searches through the readers to find the oldest point,
632  // without moving oldestUnconsumedCursor. Note that readers can continue to read while we are looping; it means
633  // oldest may not be completely accurate, but it will always be older than the readers because they are reading
634  // away from it. Also note that backwards seeks (which would break the invariant) are prevented with a mutex which
635  // is held while this function is called. Also note that all read cursors may be in the future, so we start with
636  // an unlimited barrier and work back from there.
637  Index oldest = std::numeric_limits<Index>::max();
638  for (size_t id = 0; id < header->maxReaders; ++id) {
639  // Note that this code is calling isReaderEnabled() without holding readerEnableMutex. On the surface, this
640  // appears to be a race condition because a reader may be disabled and/or re-enabled before the subsequent code
641  // reads the cursor, but it turns out to be safe because:
642  // - if a reader is enabled, its cursor is valid
643  // - if a reader becomes disabled, its cursor moves to writeCursor (which will never be the oldest)
644  // - if a reader becomes re-enabled, its cursor defaults to writeCursor (which will never be the oldest)
645  // - if a reader is created that wants to be at an older index, it gets there by doing a backward seek (which
646  // is locked when this function is called)
647  if (isReaderEnabled(id) && getReaderCursorArray()[id] < oldest) {
648  oldest = getReaderCursorArray()[id];
649  }
650  }
651 
652  // If no barrier was found, block at the write cursor so that we retain data until a reader comes along to read it.
653  if (std::numeric_limits<Index>::max() == oldest) {
654  oldest = header->writeStartCursor;
655  }
656 
657  // Now that we've measured the oldest cursor, we can safely update oldestUnconsumedCursor with no risk of an
658  // overrun of any readers.
659 
660  // To clarify the logic here, the code above reviewed all of the enabled readers to see where the oldest cursor is
661  // at. This value is captured in the 'oldest' variable. Now we want to move up our writer barrier
662  // ('oldestUnconsumedCursor') if it is older than it needs to be.
663  if (oldest > header->oldestUnconsumedCursor) {
664  header->oldestUnconsumedCursor = oldest;
665 
666  // Notify the writer(s).
667  // Note: as an optimization, we could skip this if there are no blocking writers (ACSDK-251).
668  header->spaceAvailableConditionVariable.notify_all();
669  }
670 }

672 template <typename T>
673 uint32_t SharedDataStream<T>::BufferLayout::stableHash(const char* string) {
674  // Simple, stable hash which XORs all bytes of string into the hash value.
675  uint32_t hashed = 0;
676  size_t pos = 0;
677  while (*string) {
678  hashed ^= *string << ((pos % sizeof(uint32_t)) * 8);
679  ++string;
680  ++pos;
681  }
682  return hashed;
683 }

709 template <typename T>
710 void SharedDataStream<T>::BufferLayout::calculateAndCacheConstants(size_t wordSize, size_t maxReaders) {
711  auto buffer = reinterpret_cast<uint8_t*>(m_buffer->data());
712  m_readerEnabledArray = reinterpret_cast<AtomicBool*>(buffer + calculateReaderEnabledArrayOffset());
713  m_readerCursorArray = reinterpret_cast<AtomicIndex*>(buffer + calculateReaderCursorArrayOffset(maxReaders));
714  m_readerCloseIndexArray = reinterpret_cast<AtomicIndex*>(buffer + calculateReaderCloseIndexArrayOffset(maxReaders));
715  m_dataSize = (m_buffer->size() - calculateDataOffset(wordSize, maxReaders)) / wordSize;
716  m_data = buffer + calculateDataOffset(wordSize, maxReaders);
717 }
