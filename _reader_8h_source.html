223 template <typename T>
224 SharedDataStream<T>::Reader::Reader(Policy policy, std::shared_ptr<BufferLayout> bufferLayout, uint8_t id) :
225  m_policy{policy},
226  m_bufferLayout{bufferLayout},
227  m_id{id},
228  m_readerCursor{&m_bufferLayout->getReaderCursorArray()[m_id]},
229  m_readerCloseIndex{&m_bufferLayout->getReaderCloseIndexArray()[m_id]} {
230  // Note - SharedDataStream::createReader() holds readerEnableMutex while calling this function.
231  // Read new data only.
232  // Note: It is important that new readers start with their cursor at the writer. This allows
233  // updateOldestUnconsumedCursor() to be thread-safe without holding readerEnableMutex. See
234  // updateOldestUnconsumedCursor() comments for further explanation.
235  *m_readerCursor = m_bufferLayout->getHeader()->writeStartCursor.load();
236 
237  // Read indefinitely.
238  *m_readerCloseIndex = std::numeric_limits<Index>::max();
239 
240  m_bufferLayout->enableReaderLocked(m_id);
241 }

243 template <typename T>
244 SharedDataStream<T>::Reader::~Reader() {
245  // Note: It is important that deleted readers do not change their cursor. This allows
246  // updateOldestUnconsumedCursor() to be thread-safe without holding readerEnableMutex. See
247  // updateOldestUnconsumedCursor() comments for further explanation.
248  std::lock_guard<Mutex> lock(m_bufferLayout->getHeader()->readerEnableMutex);
249  m_bufferLayout->disableReaderLocked(m_id);
250  m_bufferLayout->updateOldestUnconsumedCursor();
251 }

253 template <typename T>
254 ssize_t SharedDataStream<T>::Reader::read(void* buf, size_t nWords, std::chrono::milliseconds timeout) {
255  if (nullptr == buf) {
256  logger::acsdkError(logger::LogEntry(TAG, "readFailed").d("reason", "nullBuffer"));
257  return Error::INVALID;
258  }
259 
260  if (0 == nWords) {
261  logger::acsdkError(logger::LogEntry(TAG, "readFailed").d("reason", "invalidNumWords").d("numWords", nWords));
262  return Error::INVALID;
263  }
264 
265  // Check if closed.
266  auto readerCloseIndex = m_readerCloseIndex->load();
267  if (*m_readerCursor >= readerCloseIndex) {
268  return Error::CLOSED;
269  }
270 
271  // Initial check for overrun.
272  auto header = m_bufferLayout->getHeader();
273  if ((header->writeEndCursor - *m_readerCursor) > m_bufferLayout->getDataSize()) {
274  return Error::OVERRUN;
275  }
276 
277  // Figure out how much we can actually copy.
278  std::unique_lock<Mutex> lock(header->dataAvailableMutex, std::defer_lock);
279  if (Policy::BLOCKING == m_policy) {
280  lock.lock();
281  }
282  size_t wordsAvailable = tell(Reference::BEFORE_WRITER);
283  if (0 == wordsAvailable) {
284  if (header->writeEndCursor > 0 && !header->isWriterEnabled) {
285  return Error::CLOSED;
286  } else if (Policy::NONBLOCKING == m_policy) {
287  return Error::WOULDBLOCK;
288  } else if (Policy::BLOCKING == m_policy) {
289  if (std::chrono::milliseconds::zero() == timeout) {
290  header->dataAvailableConditionVariable.wait(
291  lock, [this] { return tell(Reference::BEFORE_WRITER) > 0; });
292  } else {
293  if (!header->dataAvailableConditionVariable.wait_for(
294  lock, timeout, [this] { return tell(Reference::BEFORE_WRITER) > 0; })) {
295  return Error::TIMEDOUT;
296  }
297  }
298  }
299  wordsAvailable = tell(Reference::BEFORE_WRITER);
300  }
301  if (Policy::BLOCKING == m_policy) {
302  lock.unlock();
303  }
304  if (nWords > wordsAvailable) {
305  nWords = wordsAvailable;
306  }
307 
308  // Don't read beyond our close index.
309  if ((*m_readerCursor + nWords) > readerCloseIndex) {
310  nWords = readerCloseIndex - *m_readerCursor;
311  }
312 
313  // Split it across the wrap.
314  size_t beforeWrap = m_bufferLayout->wordsUntilWrap(*m_readerCursor);
315  if (beforeWrap > nWords) {
316  beforeWrap = nWords;
317  }
318  size_t afterWrap = nWords - beforeWrap;
319 
320  // Copy the two segments.
321  auto buf8 = static_cast<uint8_t*>(buf);
322  memcpy(buf8, m_bufferLayout->getData(*m_readerCursor), beforeWrap * getWordSize());
323  if (afterWrap > 0) {
324  memcpy(
325  buf8 + (beforeWrap * getWordSize()),
326  m_bufferLayout->getData(*m_readerCursor + beforeWrap),
327  afterWrap * getWordSize());
328  }
329 
330  // Advance the read cursor.
331  *m_readerCursor += nWords;
332 
333  // Final check for overrun (do this before the updateOldestUnconsumedCursor() call below for improved accuracy).
334  bool overrun = ((header->writeEndCursor - *m_readerCursor) > m_bufferLayout->getDataSize());
335 
336  // Move the unconsumed cursor before returning.
337  m_bufferLayout->updateOldestUnconsumedCursor();
338 
339  // Now we can safely error out if there was an overrun.
340  if (overrun) {
341  return Error::OVERRUN;
342  }
343 
344  return nWords;
345 }

347 template <typename T>
348 bool SharedDataStream<T>::Reader::seek(Index offset, Reference reference) {
349  auto header = m_bufferLayout->getHeader();
350  auto writeStartCursor = &header->writeStartCursor;
351  auto writeEndCursor = &header->writeEndCursor;
352  Index absolute = std::numeric_limits<Index>::max();
353 
354  switch (reference) {
355  case Reference::AFTER_READER:
356  absolute = *m_readerCursor + offset;
357  break;
358  case Reference::BEFORE_READER:
359  if (offset > *m_readerCursor) {
360  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
361  .d("reason", "seekBeforeStreamStart")
362  .d("reference", "BEFORE_READER")
363  .d("seekOffset", offset)
364  .d("readerCursor", m_readerCursor->load()));
365  return false;
366  }
367  absolute = *m_readerCursor - offset;
368  break;
369  case Reference::BEFORE_WRITER:
370  if (offset > *writeStartCursor) {
371  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
372  .d("reason", "seekBeforeStreamStart")
373  .d("reference", "BEFORE_WRITER")
374  .d("seekOffset", offset)
375  .d("writeStartCursor", writeStartCursor->load()));
376  return false;
377  }
378  absolute = *writeStartCursor - offset;
379  break;
380  case Reference::ABSOLUTE:
381  absolute = offset;
382  }
383 
384  // Don't seek beyond the close index.
385  if (absolute > *m_readerCloseIndex) {
386  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
387  .d("reason", "seekBeyondCloseIndex")
388  .d("position", absolute)
389  .d("readerCloseIndex", m_readerCloseIndex->load()));
390  return false;
391  }
392 
393  // Don't seek to future indices that have not been written yet.
394  if (absolute > *writeStartCursor) {
395  logger::acsdkError(logger::LogEntry(TAG, "seekFailed")
396  .d("reason", "seekUnwrittenIndices")
397  .d("position", absolute)
398  .d("writeStartCursor", writeStartCursor->load()));
399  return false;
400  }
401 
402  // Per documentation of updateOldestUnconsumedCursor(), don't try to seek backwards while oldestConsumedCursor is
403  // being updated.
404  bool backward = absolute < *m_readerCursor;
405  std::unique_lock<Mutex> lock(header->backwardSeekMutex, std::defer_lock);
406  if (backward) {
407  lock.lock();
408  }
409 
410  // Don't seek to past data which has been (or soon will be) overwritten.
411  // Note: If this is a backward seek, it is important that this check is performed while holding the
412  // backwardSeekMutex to prevent a writer from starting to overwrite us between here and the m_readerCursor update
413  // below. If this is not a backward seek, then the mutex is not held.
414  if (*writeEndCursor - absolute > m_bufferLayout->getDataSize()) {
415  logger::acsdkError(logger::LogEntry(TAG, "seekFailed").d("reason", "seekOverwrittenData"));
416  return false;
417  }
418 
419  *m_readerCursor = absolute;
420 
421  if (backward) {
422  m_bufferLayout->updateOldestUnconsumedCursorLocked();
423  lock.unlock();
424  } else {
425  m_bufferLayout->updateOldestUnconsumedCursor();
426  }
427 
428  return true;
429 }

431 template <typename T>
432 typename SharedDataStream<T>::Index SharedDataStream<T>::Reader::tell(Reference reference) const {
433  auto writeStartCursor = &m_bufferLayout->getHeader()->writeStartCursor;
434  switch (reference) {
435  case Reference::AFTER_READER:
436  return 0;
437  case Reference::BEFORE_READER:
438  return 0;
439  case Reference::BEFORE_WRITER:
440  return *writeStartCursor - *m_readerCursor;
441  case Reference::ABSOLUTE:
442  return *m_readerCursor;
443  }
444  logger::acsdkError(logger::LogEntry(TAG, "tellFailed").d("reason", "invalidReference"));
445  return std::numeric_limits<Index>::max();
446 }

448 template <typename T>
449 void SharedDataStream<T>::Reader::close(Index offset, Reference reference) {
450  auto writeStartCursor = &m_bufferLayout->getHeader()->writeStartCursor;
451  Index absolute = 0;
452  bool validReference = false;
453  switch (reference) {
454  case Reference::AFTER_READER:
455  absolute = *m_readerCursor + offset;
456  validReference = true;
457  break;
458  case Reference::BEFORE_READER:
459  absolute = *m_readerCursor;
460  validReference = true;
461  break;
462  case Reference::BEFORE_WRITER:
463  if (*writeStartCursor < offset) {
464  logger::acsdkError(logger::LogEntry(TAG, "closeFailed")
465  .d("reason", "invalidIndex")
466  .d("reference", "BEFORE_WRITER")
467  .d("offset", offset)
468  .d("writeStartCursor", writeStartCursor->load()));
469  } else {
470  absolute = *writeStartCursor - offset;
471  }
472  validReference = true;
473  break;
474  case Reference::ABSOLUTE:
475  absolute = offset;
476  validReference = true;
477  break;
478  }
479  if (!validReference) {
480  logger::acsdkError(logger::LogEntry(TAG, "closeFailed").d("reason", "invalidReference"));
481  }
482 
483  *m_readerCloseIndex = absolute;
484 }
