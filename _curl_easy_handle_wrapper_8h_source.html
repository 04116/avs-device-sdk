240 template <typename ParamType>
241 bool CurlEasyHandleWrapper::setopt(CURLoption option, ParamType value) {
242  auto result = curl_easy_setopt(m_handle, option, value);
243  if (result != CURLE_OK) {
244  logger::acsdkError(logger::LogEntry("CurlEasyHandleWrapper", "setoptFailed")
245  .d("reason", "curl_easy_setopt failed")
246  .d("option", option)
247  .sensitive("value", value)
248  .d("result", result)
249  .d("error", curl_easy_strerror(result)));
250  return false;
251  }
252  return true;
253 }
