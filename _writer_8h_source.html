172 template <typename T>
173 ssize_t SharedDataStream<T>::Writer::write(const void* buf, size_t nWords) {
174  if (nullptr == buf) {
175  logger::acsdkError(logger::LogEntry(TAG, "writeFailed").d("reason", "nullBuffer"));
176  return Error::INVALID;
177  }
178  if (0 == nWords) {
179  logger::acsdkError(logger::LogEntry(TAG, "writeFailed").d("reason", "zeroNumWords"));
180  return Error::INVALID;
181  }
182 
183  auto header = m_bufferLayout->getHeader();
184  if (!header->isWriterEnabled) {
185  logger::acsdkError(logger::LogEntry(TAG, "writeFailed").d("reason", "writerDisabled"));
186  return Error::CLOSED;
187  }
188 
189  // Don't try to write more data than we can fit in the circular buffer.
190  if (nWords > m_bufferLayout->getDataSize()) {
191  switch (m_policy) {
192  case Policy::NONBLOCKABLE:
193  nWords = m_bufferLayout->getDataSize();
194  break;
195  case Policy::ALL_OR_NOTHING:
196  return Error::WOULDBLOCK;
197  }
198  }
199 
200  Index writeEnd = header->writeStartCursor + nWords;
201  std::unique_lock<Mutex> backwardSeekLock(header->backwardSeekMutex, std::defer_lock);
202  if (Policy::ALL_OR_NOTHING == m_policy) {
203  backwardSeekLock.lock();
204 
205  // Note - this check must be performed while locked to prevent a reader from backwards-seeking into the write
206  // region between here and the writeEndCursor update below.
207  if ((writeEnd - header->oldestUnconsumedCursor) > m_bufferLayout->getDataSize()) {
208  return Error::WOULDBLOCK;
209  }
210  }
211 
212  header->writeEndCursor = writeEnd;
213 
214  if (Policy::ALL_OR_NOTHING == m_policy) {
215  backwardSeekLock.unlock();
216  }
217 
218  // Split it across the wrap.
219  size_t beforeWrap = m_bufferLayout->wordsUntilWrap(header->writeStartCursor);
220  if (beforeWrap > nWords) {
221  beforeWrap = nWords;
222  }
223  size_t afterWrap = nWords - beforeWrap;
224 
225  // Copy the two segments.
226  auto buf8 = static_cast<const uint8_t*>(buf);
227  memcpy(m_bufferLayout->getData(header->writeStartCursor), buf8, beforeWrap * getWordSize());
228  if (afterWrap > 0) {
229  memcpy(
230  m_bufferLayout->getData(header->writeStartCursor + beforeWrap),
231  buf8 + beforeWrap * getWordSize(),
232  afterWrap * getWordSize());
233  }
234 
235  // Advance the write cursor.
236  // Note: To prevent a race condition and ensure that readers which block on dataAvailableConditionVariable don't
237  // miss a notify, we should always lock the dataAvailableConditionVariable mutex while moving writeStartCursor. As
238  // an optimization, we skip that lock for NONBLOCKABLE writers under the assumption that they will be writing
239  // continuously, so a missed notification is not significant.
240  // Note: As a further optimization, the lock could be omitted if no blocking readers are in use (ACSDK-251).
241  std::unique_lock<Mutex> dataAvailableLock(header->dataAvailableMutex, std::defer_lock);
242  if (Policy::NONBLOCKABLE == m_policy) {
243  dataAvailableLock.lock();
244  }
245  header->writeStartCursor = header->writeEndCursor.load();
246  if (Policy::NONBLOCKABLE == m_policy) {
247  dataAvailableLock.unlock();
248  }
249 
250  // Notify the reader(s).
251  // Note: as an optimization, we could skip this if there are no blocking readers (ACSDK-251).
252  header->dataAvailableConditionVariable.notify_all();
253 
254  return nWords;
255 }

262 template <typename T>
263 void SharedDataStream<T>::Writer::close() {
264  auto header = m_bufferLayout->getHeader();
265  std::lock_guard<Mutex> lock(header->writerEnableMutex);
266  if (m_closed) {
267  return;
268  }
269  if (header->isWriterEnabled) {
270  header->isWriterEnabled = false;
271  }
272  m_closed = true;
273 }
