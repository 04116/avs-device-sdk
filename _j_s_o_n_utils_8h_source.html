114 template <typename T>
115 bool retrieveValue(const rapidjson::Value& jsonNode, const std::string& key, T* value) {
116  if (!value) {
117  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "nullValue"));
118  return false;
119  }
120 
121  rapidjson::Value::ConstMemberIterator iterator;
122  if (!findNode(jsonNode, key, &iterator)) {
123  return false;
124  }
125 
126  if (!convertToValue(iterator->value, value)) {
127  return false;
128  }
129 
130  return true;
131 }

143 template <typename T>
144 bool retrieveValue(const std::string jsonString, const std::string& key, T* value) {
145  if (!value) {
146  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "nullValue"));
147  return false;

149 
150  rapidjson::Document document;

152  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "parsingError"));
153  return false;
154  }
155 
156  return retrieveValue(document, key, value);
157 }
