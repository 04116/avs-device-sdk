100 template <typename T>
101 bool retrieveValue(const rapidjson::Value& jsonNode, const std::string& key, T* value) {
102  if (!value) {
103  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "nullValue"));
104  return false;
105  }
106 
107  rapidjson::Value::ConstMemberIterator iterator;
108  if (!findNode(jsonNode, key, &iterator)) {
109  return false;
110  }
111 
112  return convertToValue(iterator->value, value);
113 }

126 template <typename T>
127 bool retrieveValue(const std::string jsonString, const std::string& key, T* value) {
128  if (!value) {
129  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "nullValue"));
130  return false;
131  }
132 
133  rapidjson::Document document;
134  if (!parseJSON(jsonString, &document)) {
135  logger::acsdkError(logger::LogEntry(getTag(), "retrieveValueFailed").d("reason", "parsingError"));
136  return false;
137  }
138 
139  return retrieveValue(document, key, value);
140 }
